def append(self, val):

    while self and self[-1] > val:
        # เรียก pop ของ parent (list) — ไม่ใช่ self.pop()

class IncreasingList(list):
    def append(self, val):
        while self and self[-1] > val:
            super().pop()
        super().append(val)

    def pop(self):
        if self:
            super().pop()

    def __len__(self):
        return super().__len__()

# return list - สร้างทั้งหมดในครั้งเดียว · กิน memory
def pages_list(items, size):
    result = []
    for i in range(0, len(items), size):
        result.append(items[i:i+size])
    return result

# yield generator - คืนทีละหน้า · ประหยัด memory
def pages_gen(items, size):
    for i in range(0, len(items), size):
        yield items[i:i+size]

def detect_high_velocity_fraud(
    transactions_df: DataFrame,
    risk_threshold: int = 300
) -> DataFrame:
    # 1. Filter
    transfers = transactions_df.filter(
        (col("transaction_type") == "TRANSFER") &
        (col("amount") >= 10000)
    )

    # 2. Window
    w = Window.partitionBy("account_id").orderBy("timestamp")

    # 3. lag
    with_prev = transfers.withColumn(
        "prev_ts", lag("timestamp").over(w)
    )

    # 4. gap + filter
    with_gap = with_prev.withColumn(
        "time_gap_s",
        unix_timestamp("timestamp") - unix_timestamp("prev_ts")
    )

    return with_gap.filter(
        col("time_gap_s").isNotNull() &
        (col("time_gap_s") <= risk_threshold)
    ).drop("prev_ts")
